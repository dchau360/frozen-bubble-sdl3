# Emscripten/WebAssembly build for Frozen Bubble SDL2
# This is a modified CMakeLists.txt for WebAssembly builds
# Use with: mkdir build-wasm && cd build-wasm && emcmake cmake -DCMAKE_TOOLCHAIN_FILE=../cmake/Emscripten.cmake .. && make

cmake_minimum_required(VERSION 3.13)

project(frozen-bubble-sdl2-wasm C CXX)

list(APPEND CMAKE_MODULE_PATH "${CMAKE_CURRENT_LIST_DIR}/cmake")

set(DEFAULT_BUILD_TYPE "Release")

# Emscripten-specific compiler flags
set(EMSCRIPTEN_FLAGS "--bind")
set(EMSCRIPTEN_FLAGS "${EMSCRIPTEN_FLAGS} -s USE_SDL=2")
set(EMSCRIPTEN_FLAGS "${EMSCRIPTEN_FLAGS} -s USE_SDL_IMAGE=2 -s SDL2_IMAGE_FORMATS=['png']")
set(EMSCRIPTEN_FLAGS "${EMSCRIPTEN_FLAGS} -s USE_SDL_MIXER=2 -s SDL2_MIXER_FORMATS=['ogg']")
set(EMSCRIPTEN_FLAGS "${EMSCRIPTEN_FLAGS} -s USE_SDL_TTF=2")
set(EMSCRIPTEN_FLAGS "${EMSCRIPTEN_FLAGS} -s ALLOW_MEMORY_GROWTH=1")
set(EMSCRIPTEN_FLAGS "${EMSCRIPTEN_FLAGS} -s DISABLE_EXCEPTION_CATCHING=0")
set(EMSCRIPTEN_FLAGS "${EMSCRIPTEN_FLAGS} -s WASM=1")
set(EMSCRIPTEN_FLAGS "${EMSCRIPTEN_FLAGS} -s ENVIRONMENT='web'")
set(EMSCRIPTEN_FLAGS "${EMSCRIPTEN_FLAGS} -s TOTAL_MEMORY=268435456")

# Asset preloading - all game assets bundled into virtual filesystem
set(EMSCRIPTEN_FLAGS "${EMSCRIPTEN_FLAGS} --preload-file ${CMAKE_SOURCE_DIR}/share@/share")

# Exported functions for JavaScript interop
set(EMSCRIPTEN_FLAGS "${EMSCRIPTEN_FLAGS} -s EXPORTED_RUNTIME_METHODS=['cwrap','ccall','setValue','getValue','UTF8ToString','stringToUTF8']")

set(CMAKE_C_FLAGS "${CMAKE_C_FLAGS} ${EMSCRIPTEN_FLAGS}")
set(CMAKE_CXX_FLAGS "${CMAKE_CXX_FLAGS} ${EMSCRIPTEN_FLAGS}")
set(CMAKE_EXE_LINKER_FLAGS "${CMAKE_EXE_LINKER_FLAGS} ${EMSCRIPTEN_FLAGS}")

# Emscripten ports are auto-linked via -s USE_* flags, no find_package needed

# iniparser is not an Emscripten port — build from bundled source
add_library(iniparser-static STATIC
    "${CMAKE_SOURCE_DIR}/third_party/iniparser/iniparser.c"
    "${CMAKE_SOURCE_DIR}/third_party/iniparser/dictionary.c"
)
target_include_directories(iniparser-static PUBLIC
    "${CMAKE_SOURCE_DIR}/third_party/iniparser"
)

add_executable(frozen-bubble-sdl2
    src/frozenbubble.cpp
    src/main.cpp
    src/menubutton.cpp
    src/mainmenu.cpp
    src/gamesettings.cpp
    src/audiomixer.cpp
    src/shaderstuff.cpp
    src/bubblegame.cpp
    src/transitionmanager.cpp
    src/ttftext.cpp
    src/highscoremanager.cpp
    src/networkclient.cpp
    src/networkclient_wasm.cpp
    src/logger.cpp
    src/platform.cpp
)

set_property(TARGET frozen-bubble-sdl2 PROPERTY CXX_STANDARD 17)
set_property(TARGET frozen-bubble-sdl2 PROPERTY CXX_EXTENSIONS OFF)

# WebAssembly doesn't use DATA_DIR - assets are preloaded
target_compile_definitions(frozen-bubble-sdl2 PRIVATE __WASM_PORT__)

target_link_libraries(frozen-bubble-sdl2 PRIVATE iniparser-static)

# Output as .html + .wasm bundle
set_target_properties(frozen-bubble-sdl2 PROPERTIES SUFFIX ".html")

message("WebAssembly build configured")
message("Assets will be preloaded from: ${CMAKE_SOURCE_DIR}/share")
